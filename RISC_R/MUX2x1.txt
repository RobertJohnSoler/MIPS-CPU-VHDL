library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity MUX2x1 is
    Port ( i0 : in STD_LOGIC;
           i1 : in STD_LOGIC;
           s : in STD_LOGIC;
           z : out STD_LOGIC);
end MUX2x1;

architecture Behavioral of MUX2x1 is
begin    
    mux2x1: process (i0, i1, s) is
    begin
        if s = '0' then
            z<=i0;
        elsif s = '1' then
            z<=i1;
        end if;
    end process mux2x1;
end Behavioral;
